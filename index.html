function importarRegistros() { const texto = document.getElementById('importArea').value.trim(); if (!texto) return; // Dividir el texto en líneas y eliminar las vacías const nuevos = texto.split('\n').filter(line => line.trim() !== ''); // Guardar en localStorage localStorage.setItem('registros', JSON.stringify(nuevos)); // Mostrar en la lista const lista = document.getElementById('registroList'); lista.innerHTML = ''; nuevos.forEach(registro => { const li = document.createElement('li'); li.textContent = registro; lista.appendChild(li); }); // Limpiar área de importación document.getElementById('importArea').value = ''; }
